# 底部进度条融合实现方案

## 设计目标

将进度条与底部状态栏（文件名显示区域）融合，实现：
1. ✅ 进度条作为状态栏的背景
2. ✅ 文字始终可见（在进度条上方）
3. ✅ 视觉统一，不突兀
4. ✅ 快速加载模式支持

##### 3. 修改 bridge.js - 添加进度条控制函数

```javascript
// ========== 底部进度条控制 ==========

const bottomProgress = {
    overlay: null,
    bar: null,
    
    init() {
        this.overlay = document.getElementById('bottomProgressOverlay');
        this.bar = this.overlay?.querySelector('.progress-bar');
    },
    
    show() {
        if (!this.overlay) this.init();
        if (this.overlay) {
            this.overlay.classList.remove('hidden');
            this.bar.style.width = '0%';
            // 强制重绘
            this.overlay.offsetHeight;
            this.bar.style.width = '10%';
        }
    },
    
    update(percent) {
        if (!this.bar) this.init();
        if (this.bar) {
            this.bar.style.width = percent + '%';
        }
    },
    
    hide() {
        if (!this.bar) this.init();
        if (this.bar && this.overlay) {
            this.bar.style.width = '100%';
            setTimeout(() => {
                this.overlay.classList.add('hidden');
                this.bar.style.width = '0%';
            }, 300);
        }
    }
};

// 初始化
document.addEventListener('DOMContentLoaded', () => {
    bottomProgress.init();
});
```

##### 4. 修改 loadPdfQuick 函数

```javascript
/**
 * 快速加载PDF（用于撤回/重做）
 */
async function loadPdfQuick(url) {
    debugLog('loadPdfQuick被调用（快速模式）');
    
    try {
        // 显示底部进度条
        bottomProgress.show();
        
        // 清空现有内容
        elements.viewer.innerHTML = '';
        
        bottomProgress.update(30);
        
        // 加载PDF
        const loadingTask = pdfjsLib.getDocument(url);
        state.pdfDoc = await loadingTask.promise;
        
        state.totalPages = state.pdfDoc.numPages;
        state.currentPage = 1;
        state.rotation = 0;
        
        bottomProgress.update(60);
        
        // 直接渲染第一页
        await renderPage(1);
        
        bottomProgress.update(90);
        
        // 后台异步更新缩略图
        updateThumbnailsInBackground();
        
        // 通知C#
        sendToHost('pdfLoaded', {
            totalPages: state.totalPages,
            currentPage: 1
        });
        
        bottomProgress.update(100);
        
        // 延迟隐藏，让用户看到完成状态
        setTimeout(() => {
            bottomProgress.hide();
        }, 200);
        
        debugLog('PDF快速加载完成');
        
    } catch (error) {
        bottomProgress.hide();
        debugLog('PDF快速加载错误: ' + error.message);
        sendToHost('loadError', { error: error.message });
    }
}

/**
 * 后台更新缩略图
 */
async function updateThumbnailsInBackground() {
    setTimeout(async () => {
        try {
            debugLog('开始后台更新缩略图...');
            await generateThumbnails();
            debugLog('缩略图后台更新完成');
        } catch (error) {
            debugLog('缩略图后台更新失败: ' + error.message);
        }
    }, 100);
}
```

```javascript
/**
 * 加载PDF文件（标准模式）
 */
async function loadPdf(url) {
    debugLog('loadPdf被调用，URL长度: ' + url.length);
    
    try {
        // 标准模式使用全屏加载动画
        showLoading('正在加载PDF文档...');

        // 清空现有内容
        elements.viewer.innerHTML = '';
        
        // 加载PDF
        const loadingTask = pdfjsLib.getDocument(url);
        state.pdfDoc = await loadingTask.promise;
        
        state.totalPages = state.pdfDoc.numPages;
        state.currentPage = 1;
        state.rotation = 0;

        // 生成缩略图
        showLoading(`正在生成缩略图 (共${state.totalPages}页)...`);
        await generateThumbnails();
        
        // 加载图层
        showLoading('正在加载图层...');
        await loadLayers();

        // 适应页面
        showLoading('正在准备显示...');
        await fitToPage();

        // 渲染第一页
        showLoading('正在渲染第一页...');
        await renderPage(1);

        // 通知C#
        sendToHost('pdfLoaded', {
            totalPages: state.totalPages,
            currentPage: 1
        });
        
        hideLoading();
        debugLog('PDF加载完成');
        
    } catch (error) {
        debugLog('PDF加载错误: ' + error.message);
        showLoading('加载失败: ' + error.message);
        setTimeout(hideLoading, 3000);
        sendToHost('loadError', { error: error.message });
    }
}
```

##### 2. 添加进度条控制方法

```csharp
/// <summary>
/// 显示底部进度条
/// </summary>
private void ShowBottomProgress()
{
    if (InvokeRequired)
    {
        Invoke(new Action(ShowBottomProgress));
        return;
    }
    
    _progressBar.Visible = true;
    _progressBar.Width = 0;
}

/// <summary>
/// 更新底部进度条
/// </summary>
private void UpdateBottomProgress(int percent)
{
    if (InvokeRequired)
    {
        Invoke(new Action<int>(UpdateBottomProgress), percent);
        return;
    }
    
    int targetWidth = (int)(statusPanel.Width * percent / 100.0);
    _progressBar.Width = targetWidth;
}

/// <summary>
/// 隐藏底部进度条
/// </summary>
private void HideBottomProgress()
{
    if (InvokeRequired)
    {
        Invoke(new Action(HideBottomProgress));
        return;
    }
    
    // 先显示100%
    _progressBar.Width = statusPanel.Width;
    
    // 延迟隐藏
    Task.Delay(300).ContinueWith(_ =>
    {
        if (InvokeRequired)
        {
            Invoke(new Action(() =>
            {
                _progressBar.Visible = false;
                _progressBar.Width = 0;
            }));
        }
        else
        {
            _progressBar.Visible = false;
            _progressBar.Width = 0;
        }
    });
}
```

##### 3. 在 ShowPdfPreview 中使用

```csharp
private void ShowPdfPreview(byte[] pdfBytes, bool quickMode = false)
{
    try
    {
        if (quickMode)
        {
            ShowBottomProgress();
            UpdateBottomProgress(10);
        }
        
        string tempPath = Path.Combine(Path.GetTempPath(), 
            $"temp_preview_{Guid.NewGuid()}.pdf");
        File.WriteAllBytes(tempPath, pdfBytes);
        
        if (quickMode)
        {
            UpdateBottomProgress(30);
        }
        
        _currentFilePath = tempPath;
        
        string loadFunction = quickMode ? "loadPdfQuick" : "loadPdf";
        string script = $"{loadFunction}('file:///{tempPath.Replace("\\", "/")}');";
        
        _ = _pdfPreview.EvaluateScriptAsync(script);
        
        if (quickMode)
        {
            UpdateBottomProgress(60);
        }
        
        // 更新检查器
        if (_inspectorForm != null && _inspectorForm.Visible)
        {
            _inspectorForm.LoadPdf(tempPath, _currentPage);
        }
        
        if (quickMode)
        {
            // 延迟隐藏进度条
            Task.Delay(500).ContinueWith(_ => HideBottomProgress());
        }
    }
    catch (Exception ex)
    {
        if (quickMode)
        {
            HideBottomProgress();
        }
        LogHelper.Error($"[PdfOperationsPanel] 显示预览失败: {ex.Message}", ex);
        ShowError($"显示预览失败: {ex.Message}");
    }
}
```
