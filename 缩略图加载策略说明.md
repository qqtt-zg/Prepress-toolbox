```javascript
// 为所有页面创建占位符
for (let pageNum = 1; pageNum <= state.totalPages; pageNum++) {

```

**占位符特点**:

- 显示页码
- 固定尺寸 (80x100px)
- 可点击跳转

**示例**:

### 阶段2: 快速加载前100页

```javascript
const INITIAL_COUNT = 100; // 初始加载100个
const initialCount = Math.min(INITIAL_COUNT, state.totalPages);

// 分批加载，每批50页
for (let batch = 0; batch < initialCount; batch += 50) {
    const batchStart = batch + 1;
    const batchEnd = Math.min(batch + 50, initialCount);
    
    // 显示进度
    elements.loadingText.textContent = 
        `正在生成缩略图 (${batchEnd}/${state.totalPages})...`;
    
    // 并行加载一批
    const batchPromises = [];
    for (let pageNum = batchStart; pageNum <= batchEnd; pageNum++) {
        batchPromises.push(loadThumbnail(pageNum, THUMBNAIL_SCALE));
    }
    
    await Promise.all(batchPromises);
}
```

**批次划分**:
- **批次1**: 第1-50页
- **批次2**: 第51-100页

**并行加载**:
- 每批内的50页并行加载
- 使用 `Promise.all()` 等待批次完成
- 批次之间串行执行

**进度显示**:
```
正在生成缩略图 (50/200)...
正在生成缩略图 (100/200)...
```

### 阶段3: 后台加载剩余页面

```javascript
if (state.totalPages > INITIAL_COUNT) {
    startBackgroundThumbnailLoading(INITIAL_COUNT + 1, THUMBNAIL_SCALE);
}
```

**后台加载特点**:
- 从第101页开始
- 异步执行，不阻塞UI
- 每10页休息50ms
- 支持优先队列（用户跳转时）

**实现**:
```javascript
function startBackgroundThumbnailLoading(startPage, scale) {
    thumbnailState.backgroundLoading = true;
    
    (async () => {
        for (let pageNum = startPage; pageNum <= state.totalPages; pageNum++) {
            // 检查优先队列
            while (thumbnailState.priorityQueue.length > 0) {
                const priorityPage = thumbnailState.priorityQueue.shift();
                await loadThumbnail(priorityPage, scale);
            }
            
            // 加载当前页
            if (!thumbnailState.loaded.has(pageNum)) {
                await loadThumbnail(pageNum, scale);
            }
            
            // 每10个休息一下
            if (pageNum % 10 === 0) {
                await new Promise(resolve => setTimeout(resolve, 50));
            }
        }
        
        thumbnailState.backgroundLoading = false;
    })();
}
```

## 优先加载机制

### 用户跳转触发

当用户跳转到某一页时，会触发优先加载：

```javascript
// 在 renderPage() 中调用
priorityLoadThumbnail(state.currentPage);
```

### 优先队列

```javascript
async function priorityLoadThumbnail(pageNum) {
    if (thumbnailState.loaded.has(pageNum)) {
        return; // 已加载
    }
    
    if (!thumbnailState.loading.has(pageNum)) {
        // 插队到最前面
        if (!thumbnailState.priorityQueue.includes(pageNum)) {
            thumbnailState.priorityQueue.unshift(pageNum);
            debugLog(`优先加载缩略图: ${pageNum}`);
        }
    }
}
```

**工作原理**:
1. 用户跳转到第150页
2. 检查第150页缩略图是否已加载
3. 如果未加载，插入优先队列最前面
4. 后台加载线程检查优先队列
5. 优先处理队列中的页面
6. 继续正常的顺序加载

## 缩略图状态管理

```javascript
const thumbnailState = {
    loaded: new Set(),           // 已加载的页面
    loading: new Set(),          // 正在加载的页面
    priorityQueue: [],           // 优先加载队列
    backgroundLoading: false     // 是否正在后台加载
};
```

### 状态转换

```
未加载 (占位符)
    ↓
正在加载 (loading.add)
    ↓
已加载 (loaded.add, loading.delete)
```

## 优化策略

### 1. 批量并行加载

```javascript
// 每批50页并行加载
const batchPromises = [];
for (let pageNum = batchStart; pageNum <= batchEnd; pageNum++) {
    batchPromises.push(loadThumbnail(pageNum, THUMBNAIL_SCALE));
}
await Promise.all(batchPromises);
```

**优点**:
- 充分利用多核CPU
- 减少总加载时间
- 提升用户体验

### 2. 后台异步加载

```javascript
// 不阻塞主线程
(async () => {
    for (let pageNum = startPage; pageNum <= state.totalPages; pageNum++) {
        await loadThumbnail(pageNum, scale);
        
        // 每10个休息一下
        if (pageNum % 10 === 0) {
            await new Promise(resolve => setTimeout(resolve, 50));
        }
    }
})();
```

**优点**:
- 不影响用户操作
- 避免UI卡顿
- 渐进式加载

### 3. 优先队列

```javascript
// 用户跳转时优先加载
while (thumbnailState.priorityQueue.length > 0) {
    const priorityPage = thumbnailState.priorityQueue.shift();
    await loadThumbnail(priorityPage, scale);
}
```

**优点**:
- 响应用户操作
- 提升感知性能
- 智能资源分配

### 4. 缩略图缩放

```javascript
const THUMBNAIL_SCALE = 0.1; // 10%的原始尺寸
```

**优点**:
- 减少渲染时间
- 降低内存占用
- 足够清晰识别

## 常见问题

### Q1: 为什么选择100页作为初始加载数量？

**A**: 基于以下考虑：
- **用户体验**: 3秒左右的等待时间可接受
- **覆盖率**: 大多数PDF在100页以内
- **性能平衡**: 不会占用过多内存
- **响应速度**: 用户可以快速开始浏览

### Q2: 可以增加初始加载数量吗？

**A**: 可以，但需要权衡：
- **优点**: 更多页面立即可用
- **缺点**: 初始加载时间更长
- **建议**: 根据目标用户的PDF大小调整

### Q3: 后台加载会影响性能吗？

**A**: 影响很小：
- 每10页休息50ms
- 不阻塞主线程
- 优先响应用户操作
- 可以随时暂停

### Q4: 优先队列如何工作？

**A**: 
1. 用户跳转到某页
2. 检查该页缩略图是否已加载
3. 如果未加载，插入优先队列最前面
4. 后台加载线程优先处理队列中的页面
5. 处理完优先队列后继续正常加载

### Q5: 如何查看当前加载进度？

**A**: 
- 查看控制台日志
- 观察缩略图面板（占位符 → 实际缩略图）
- 加载提示文本显示进度
