**执行流程**：
1. `ValidateInput()` → 验证输入文件

```

纸张2 正面: [14][3]  反面: [4][13]
纸张3 正面: [12][5]  反面: [6][11]
纸张4 正面: [10][7]  反面: [8][9]
```

## 6. 开发指南：扩展新拼版模式

### 6.1 添加新服务的步骤

以 `NUpService` 为例：

**步骤 1：创建服务类**

```csharp
// src/WindowsFormsApp3/Services/NUpService.cs
using WindowsFormsApp3.Interfaces;
using WindowsFormsApp3.Models;

namespace WindowsFormsApp3.Services
{
    public class NUpService : ImpositionBase
    {
        public NUpService(IPdfInfoProvider pdfInfoProvider = null)
            : base(pdfInfoProvider)
        {
        }

        // 1. 定义配置类
        public class NUpConfig
        {
            public string InputPdfPath { get; set; }
            public string OutputPdfPath { get; set; }
            public float PaperWidth { get; set; }   // mm
            public float PaperHeight { get; set; }  // mm
            public int Rows { get; set; }           // 0=自动
            public int Columns { get; set; }        // 0=自动
            public float Gutter { get; set; }       // 间距 mm
            public float Bleed { get; set; }        // 出血 mm
            public bool AutoRotate { get; set; }
            public int CopyCount { get; set; }      // 一式N联
        }

        // 2. 实现主方法
        public async Task<string> GenerateNUpPdfAsync(
            NUpConfig config,
            IProgress<int> progress = null,
            CancellationToken cancellationToken = default)
        {
            // 验证
            ValidateInput(config.InputPdfPath);
            ReportProgress(progress, 5);

            // 分析 PDF
            var pdfInfo = await AnalyzePdfAsync(config.InputPdfPath, cancellationToken);
            ReportProgress(progress, 15);

            // 计算布局（调用 ImpositionService）
            var impositionService = new ImpositionService(_pdfInfoProvider);
            var flatConfig = new FlatSheetConfiguration
            {
                PaperWidth = config.PaperWidth,
                PaperHeight = config.PaperHeight,
                Rows = config.Rows,
                Columns = config.Columns,
                CopyCount = config.CopyCount
            };
            var pdfInfoModel = ImpositionPdfInfo.FromPdfFileInfo(
                pdfInfo,
                (float)(pdfInfo.FirstPageSize?.Width ?? 0),
                (float)(pdfInfo.FirstPageSize?.Height ?? 0),
                false);
            var layoutResult = await impositionService.CalculateFlatSheetLayoutAsync(
                flatConfig, pdfInfoModel, cancellationToken);
            ReportProgress(progress, 30);

            // 生成 PDF
            var outputPath = string.IsNullOrEmpty(config.OutputPdfPath)
                ? GenerateOutputPath(config.InputPdfPath, "nup")
                : config.OutputPdfPath;

            await GeneratePdfAsync(config, layoutResult, pdfInfo, outputPath, progress, cancellationToken);
            ReportProgress(progress, 100);

            return outputPath;
        }

        // 3. 实现 PDF 生成逻辑
        private async Task GeneratePdfAsync(
            NUpConfig config,
            ImpositionResult layoutResult,
            PdfFileInfo pdfInfo,
            string outputPath,
            IProgress<int> progress,
            CancellationToken cancellationToken)
        {
            await Task.Run(() =>
            {
                using (var srcPdf = new PdfDocument(new PdfReader(config.InputPdfPath)))
                using (var destPdf = new PdfDocument(new PdfWriter(outputPath)))
                {
                    var pageSize = new iText.Kernel.Geom.PageSize(
                        config.PaperWidth * 72f / 25.4f,
                        config.PaperHeight * 72f / 25.4f);

                    int totalPages = pdfInfo.PageCount ?? 0;
                    int pagesPerSheet = layoutResult.Rows * layoutResult.Columns;

                    for (int i = 1; i <= totalPages; i += pagesPerSheet)
                    {
                        cancellationToken.ThrowIfCancellationRequested();

                        var page = destPdf.AddNewPage(pageSize);
                        var canvas = new PdfCanvas(page);

                        // 按行列放置页面
                        for (int row = 0; row < layoutResult.Rows; row++)
                        {
                            for (int col = 0; col < layoutResult.Columns; col++)
                            {
                                int pageNum = i + row * layoutResult.Columns + col;
                                if (pageNum > totalPages) break;

                                float x = col * layoutResult.CellWidth * 72f / 25.4f;
                                float y = (layoutResult.Rows - 1 - row) * layoutResult.CellHeight * 72f / 25.4f;

                                PlacePageOnCanvas(canvas, srcPdf, pageNum,
                                    x, y,
                                    layoutResult.CellWidth * 72f / 25.4f,
                                    layoutResult.CellHeight * 72f / 25.4f);
                            }
                        }

                        ReportProgress(progress, 30 + (int)(i * 70.0 / totalPages));
                    }
                }
            }, cancellationToken);
        }

        // 4. 复用 SaddleStitchService 中的 PlacePageOnCanvas 方法
        private void PlacePageOnCanvas(
            PdfCanvas canvas, PdfDocument srcPdf,
            int pageNum, float x, float y, float width, float height)
        {
            var srcPage = srcPdf.GetPage(pageNum);
            var pageCopy = srcPage.CopyAsFormXObject(canvas.GetDocument());
            var srcSize = srcPage.GetPageSize();
            var scaleX = width / srcSize.GetWidth();
            var scaleY = height / srcSize.GetHeight();
            var scale = Math.Min(scaleX, scaleY);
            var scaledWidth = srcSize.GetWidth() * scale;
            var scaledHeight = srcSize.GetHeight() * scale;
            var offsetX = (width - scaledWidth) / 2;
            var offsetY = (height - scaledHeight) / 2;

            canvas.AddXObjectFittedIntoRectangle(pageCopy,
                new iText.Kernel.Geom.Rectangle(x + offsetX, y + offsetY, scaledWidth, scaledHeight));
        }
    }
}
```

### 6.3 iText 常用 API 速查

```csharp
// 打开/创建 PDF
var srcPdf = new PdfDocument(new PdfReader(inputPath));
var destPdf = new PdfDocument(new PdfWriter(outputPath));

// 添加页面
var page = destPdf.AddNewPage(new PageSize(width, height));

// 在画布上绘制
var canvas = new PdfCanvas(page);
canvas.SaveState();
canvas.SetStrokeColor(DeviceCmyk.BLACK);
canvas.SetLineWidth(0.5f);
canvas.MoveTo(x1, y1);
canvas.LineTo(x2, y2);
canvas.Stroke();
canvas.RestoreState();

// 复制源页面到目标位置
var srcPage = srcPdf.GetPage(pageNum);
var formXObject = srcPage.CopyAsFormXObject(destPdf);
canvas.AddXObjectFittedIntoRectangle(formXObject,
    new Rectangle(x, y, width, height));

// 获取页面尺寸
var size = page.GetPageSize();
float widthPt = size.GetWidth();   // 点（1pt = 1/72 inch）
float heightPt = size.GetHeight();

// 单位转换
float mmToPt = mm * 72f / 25.4f;
float ptToMm = pt * 25.4f / 72f;
```
