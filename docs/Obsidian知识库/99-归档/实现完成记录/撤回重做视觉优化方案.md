# 撤回/重做视觉优化方案

## 问题分析

### 当前问题
用户在执行撤回/重做操作时，会看到完整的加载动画，导致：
1. ❌ 无法清晰看到内容变化
2. ❌ 感觉像是重新加载文件
3. ❌ 操作反馈不够直观
4. ❌ 影响快速连续操作

### 根本原因
```javascript
// 当前实现：每次加载都显示完整动画
async function loadPdf(url) {
    showLoading('正在加载PDF文档...');      // ← 阻塞式加载动画
    // ... 加载PDF
    showLoading('正在生成缩略图...');        // ← 阻塞式加载动画
    // ... 生成缩略图
    showLoading('正在加载图层...');          // ← 阻塞式加载动画
    // ... 加载图层
    hideLoading();
}
```

## 优化方案

### 方案1: 快速切换模式（推荐）⭐

#### 核心思路
撤回/重做时跳过耗时的初始化步骤，直接渲染页面。

#### 实现方案

##### 1.1 添加快速加载函数

```javascript
/**
 * 快速加载PDF（用于撤回/重做）
 * 跳过缩略图和图层生成，直接渲染
 */
async function loadPdfQuick(url) {
    debugLog('loadPdfQuick被调用（快速模式）');
    
    try {
        // 不显示加载动画，只显示轻量提示
        showQuickLoadingHint();
        
        // 清空现有内容
        elements.viewer.innerHTML = '';
        
        // 加载PDF
        const loadingTask = pdfjsLib.getDocument(url);
        state.pdfDoc = await loadingTask.promise;
        
        state.totalPages = state.pdfDoc.numPages;
        state.currentPage = 1;
        state.rotation = 0;
        
        // 直接渲染第一页，不生成缩略图
        await renderPage(1);
        
        // 后台异步更新缩略图（不阻塞）
        updateThumbnailsInBackground();
        
        // 通知C#
        sendToHost('pdfLoaded', {
            totalPages: state.totalPages,
            currentPage: 1
        });
        
        hideQuickLoadingHint();
        debugLog('PDF快速加载完成');
        
    } catch (error) {
        debugLog('PDF快速加载错误: ' + error.message);
        showError('加载失败: ' + error.message);
    }
}

/**
 * 显示轻量加载提示（不阻塞视图）
 */
function showQuickLoadingHint() {
    // 在右上角显示小型加载指示器
    const hint = document.createElement('div');
    hint.id = 'quickLoadingHint';
    hint.style.cssText = `
        position: fixed;
        top: 70px;
        right: 20px;
        background: rgba(0, 0, 0, 0.8);
        color: #fff;
        padding: 8px 16px;
        border-radius: 4px;
        font-size: 12px;
        z-index: 200;
        display: flex;
        align-items: center;
        gap: 8px;
    `;
    hint.innerHTML = `
        <div class="mini-spinner"></div>
        <span>加载中...</span>
    `;
    document.body.appendChild(hint);
}

function hideQuickLoadingHint() {
    const hint = document.getElementById('quickLoadingHint');
    if (hint) {
        hint.remove();
    }
}

/**
 * 后台更新缩略图（不阻塞）
 */
async function updateThumbnailsInBackground() {
    // 异步更新，不等待
    setTimeout(async () => {
        try {
            await generateThumbnails();
            debugLog('缩略图后台更新完成');
        } catch (error) {
            debugLog('缩略图后台更新失败: ' + error.message);
        }
    }, 100);
}
```

### 方案2: 淡入淡出过渡效果

#### 核心思路
使用CSS过渡效果，让内容变化更平滑可见。

#### 实现方案

##### 2.1 添加CSS过渡

```css
/* 在 viewer.html 的 <style> 中添加 */

/* PDF查看器容器过渡 */
#viewer {
    transition: opacity 0.3s ease-in-out;
}

#viewer.transitioning {
    opacity: 0.3;
}

/* 轻量加载提示 */
.mini-spinner {
    width: 16px;
    height: 16px;
    border: 2px solid rgba(255, 255, 255, 0.3);
    border-top-color: #fff;
    border-radius: 50%;
    animation: spin 0.8s linear infinite;
}

@keyframes spin {
    to { transform: rotate(360deg); }
}
```

##### 2.2 修改渲染逻辑

```javascript
/**
 * 渲染页面（带过渡效果）
 */
async function renderPageWithTransition(pageNum) {
    // 添加过渡类
    elements.viewer.classList.add('transitioning');
    
    // 等待过渡开始
    await new Promise(resolve => setTimeout(resolve, 50));
    
    // 渲染新内容
    await renderPage(pageNum);
    
    // 移除过渡类
    elements.viewer.classList.remove('transitioning');
}
```

### 方案3: 保留上一帧 + 覆盖新内容

#### 核心思路
在新内容加载时保留旧内容，新内容准备好后直接替换。

#### 实现方案

```javascript
/**
 * 无缝切换PDF内容
 */
async function switchPdfContent(url) {
    debugLog('switchPdfContent被调用（无缝模式）');
    
    try {
        // 保存当前显示的内容
        const oldContent = elements.viewer.innerHTML;
        
        // 在后台加载新PDF
        const loadingTask = pdfjsLib.getDocument(url);
        const newPdfDoc = await loadingTask.promise;
        
        // 在内存中渲染第一页
        const page = await newPdfDoc.getPage(1);
        const viewport = page.getViewport({ scale: state.scale });
        
        // 创建临时Canvas
        const tempCanvas = document.createElement('canvas');
        const context = tempCanvas.getContext('2d');
        tempCanvas.width = viewport.width;
        tempCanvas.height = viewport.height;
        
        // 渲染到临时Canvas
        await page.render({
            canvasContext: context,
            viewport: viewport
        }).promise;
        
        // 更新状态
        state.pdfDoc = newPdfDoc;
        state.totalPages = newPdfDoc.numPages;
        state.currentPage = 1;
        
        // 快速替换内容（几乎瞬间）
        elements.viewer.innerHTML = '';
        tempCanvas.className = 'page';
        elements.viewer.appendChild(tempCanvas);
        
        // 更新UI
        updateUI();
        
        // 后台更新缩略图
        updateThumbnailsInBackground();
        
        debugLog('PDF内容无缝切换完成');
        
    } catch (error) {
        debugLog('PDF切换错误: ' + error.message);
        // 恢复旧内容
        elements.viewer.innerHTML = oldContent;
    }
}
```

### 方案4: 进度条替代遮罩（最小干扰）

#### 核心思路
用顶部进度条替代全屏遮罩，用户仍能看到内容变化。

#### 实现方案

##### 4.1 添加进度条HTML

```html
<!-- 在 viewer.html 的 <body> 开头添加 -->
<div id="topProgressBar" class="hidden">
    <div class="progress-fill"></div>
</div>
```

##### 4.2 添加进度条CSS

```css
#topProgressBar {
    position: fixed;
    top: 0;
    left: 0;
    right: 0;
    height: 3px;
    background: rgba(0, 0, 0, 0.1);
    z-index: 1000;
    overflow: hidden;
}

#topProgressBar.hidden {
    display: none;
}

.progress-fill {
    height: 100%;
    background: linear-gradient(90deg, #4a9eff, #1890ff);
    width: 0%;
    transition: width 0.3s ease;
    animation: progress-shimmer 1.5s infinite;
}

@keyframes progress-shimmer {
    0% { opacity: 1; }
    50% { opacity: 0.7; }
    100% { opacity: 1; }
}
```

##### 4.3 修改加载逻辑

```javascript
function showTopProgress() {
    const bar = document.getElementById('topProgressBar');
    const fill = bar.querySelector('.progress-fill');
    bar.classList.remove('hidden');
    fill.style.width = '30%';
}

function updateTopProgress(percent) {
    const fill = document.querySelector('.progress-fill');
    fill.style.width = percent + '%';
}

function hideTopProgress() {
    const bar = document.getElementById('topProgressBar');
    const fill = bar.querySelector('.progress-fill');
    fill.style.width = '100%';
    setTimeout(() => {
        bar.classList.add('hidden');
        fill.style.width = '0%';
    }, 300);
}

// 在快速加载中使用
async function loadPdfQuick(url) {
    showTopProgress();
    
    try {
        updateTopProgress(30);
        const loadingTask = pdfjsLib.getDocument(url);
        state.pdfDoc = await loadingTask.promise;
        
        updateTopProgress(60);
        await renderPage(1);
        
        updateTopProgress(100);
        hideTopProgress();
    } catch (error) {
        hideTopProgress();
        showError(error.message);
    }
}
```
