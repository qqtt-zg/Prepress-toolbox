# PDF操作面板 - 专业撤回/重做系统设计

## 设计目标

实现类似 Adobe Acrobat DC 和 Enfocus PitStop Pro 的专业撤回/重做功能：
- 支持多级撤回/重做
- 撤回后重做按钮立即可用
- 使用历史栈管理操作状态
- 新操作自动清除重做历史

## 当前问题分析

### 现有实现的局限性
1. **单次撤回/重做** - 只能撤回最后一次转曲操作
2. **基于文件加载** - 通过重新加载文件实现撤回，效率较低
3. **重做按钮逻辑错误** - 代码中撤回后应启用重做，但描述说是灰色的
4. **无历史记录** - 无法查看操作历史或跳转到特定状态

### 专业软件的优势
- **历史栈机制** - 维护完整的操作历史
- **差异存储** - 只存储变化，节省内存
- **即时响应** - 撤回/重做立即生效
- **可视化历史** - 可选的历史面板显示所有操作

## 设计方案

### 方案一：完整历史栈系统（推荐用于未来扩展）

#### 数据结构
```csharp
/// <summary>
/// PDF操作历史记录项
/// </summary>
public class PdfOperationHistoryItem
{
    public string OperationType { get; set; }      // 操作类型：转曲、裁切标记、套准标记等
    public DateTime Timestamp { get; set; }        // 操作时间
    public byte[] PdfBytes { get; set; }           // PDF字节数组
    public string Description { get; set; }        // 操作描述
    public Dictionary<string, object> Metadata { get; set; } // 元数据
}

/// <summary>
/// PDF操作历史管理器
/// </summary>
public class PdfOperationHistoryManager
{
    private Stack<PdfOperationHistoryItem> _undoStack;
    private Stack<PdfOperationHistoryItem> _redoStack;
    private int _maxHistorySize = 20; // 最大历史记录数
    
    public bool CanUndo => _undoStack.Count > 0;
    public bool CanRedo => _redoStack.Count > 0;
    public int UndoCount => _undoStack.Count;
    public int RedoCount => _redoStack.Count;
    
    /// <summary>
    /// 添加新操作到历史栈
    /// </summary>
    public void PushOperation(PdfOperationHistoryItem item)
    {
        _undoStack.Push(item);
        _redoStack.Clear(); // 新操作清除重做栈
        
        // 限制历史栈大小
        if (_undoStack.Count > _maxHistorySize)
        {
            var items = _undoStack.ToList();
            items.RemoveAt(items.Count - 1); // 移除最旧的
            _undoStack = new Stack<PdfOperationHistoryItem>(items.Reverse<PdfOperationHistoryItem>());
        }
    }
    
    /// <summary>
    /// 撤回操作
    /// </summary>
    public PdfOperationHistoryItem Undo()
    {
        if (!CanUndo) return null;
        
        var item = _undoStack.Pop();
        _redoStack.Push(item);
        
        // 返回撤回后应该显示的状态（栈顶或原始状态）
        return _undoStack.Count > 0 ? _undoStack.Peek() : null;
    }
    
    /// <summary>
    /// 重做操作
    /// </summary>
    public PdfOperationHistoryItem Redo()
    {
        if (!CanRedo) return null;
        
        var item = _redoStack.Pop();
        _undoStack.Push(item);
        
        return item;
    }
    
    /// <summary>
    /// 清除所有历史
    /// </summary>
    public void Clear()
    {
        _undoStack.Clear();
        _redoStack.Clear();
    }
    
    /// <summary>
    /// 获取撤回历史列表（用于UI显示）
    /// </summary>
    public List<string> GetUndoHistory()
    {
        return _undoStack.Select(x => x.Description).ToList();
    }
    
    /// <summary>
    /// 获取重做历史列表（用于UI显示）
    /// </summary>
    public List<string> GetRedoHistory()
    {
        return _redoStack.Select(x => x.Description).ToList();
    }
}
```

#### 集成到 PdfOperationsPanel
```csharp
public partial class PdfOperationsPanel : BasePanelControl
{
    private PdfOperationHistoryManager _historyManager;
    private string _originalFilePath;
    private byte[] _originalPdfBytes; // 保存原始PDF字节
    
    public PdfOperationsPanel()
    {
        InitializeComponent();
        _historyManager = new PdfOperationHistoryManager();
    }
    
    /// <summary>
    /// 加载PDF文件
    /// </summary>
    private async Task LoadPdfFileAsync(string filePath)
    {
        _originalFilePath = filePath;
        _originalPdfBytes = File.ReadAllBytes(filePath);
        
        // 清除历史
        _historyManager.Clear();
        UpdateHistoryButtons();
        
        // 加载预览
        await _pdfPreview.LoadPdfAsync(filePath);
    }
    
    /// <summary>
    /// 处理字体转曲完成事件
    /// </summary>
    private void Inspector_FontOutlineCompleted(object sender, FontOutlineCompletedEventArgs e)
    {
        // 添加到历史栈
        var historyItem = new PdfOperationHistoryItem
        {
            OperationType = "FontOutline",
            Timestamp = DateTime.Now,
            PdfBytes = e.PdfBytes,
            Description = "字体转曲",
            Metadata = new Dictionary<string, object>
            {
                { "FontCount", e.FontCount },
                { "FileName", Path.GetFileName(_originalFilePath) }
            }
        };
        
        _historyManager.PushOperation(historyItem);
        UpdateHistoryButtons();
        
        // 显示转曲后的PDF
        ShowPdfPreview(e.PdfBytes);
        
        UpdateStatus($"已转曲 (未保存) - {Path.GetFileName(_originalFilePath)}");
    }
    
    /// <summary>
    /// 撤回操作
    /// </summary>
    private void BtnUndo_Click(object sender, EventArgs e)
    {
        if (!_historyManager.CanUndo)
        {
            ShowWarning("没有可以撤回的操作");
            return;
        }
        
        var previousState = _historyManager.Undo();
        UpdateHistoryButtons();
        
        if (previousState != null)
        {
            // 显示上一个状态
            ShowPdfPreview(previousState.PdfBytes);
            UpdateStatus($"已撤回到: {previousState.Description}");
        }
        else
        {
            // 回到原始状态
            ShowPdfPreview(_originalPdfBytes);
            UpdateStatus($"已加载: {Path.GetFileName(_originalFilePath)}");
        }
        
        LogHelper.Info($"[PdfOperationsPanel] 撤回操作，剩余历史: {_historyManager.UndoCount}");
    }
    
    /// <summary>
    /// 重做操作
    /// </summary>
    private void BtnRedo_Click(object sender, EventArgs e)
    {
        if (!_historyManager.CanRedo)
        {
            ShowWarning("没有可以重做的操作");
            return;
        }
        
        var redoState = _historyManager.Redo();
        UpdateHistoryButtons();
        
        if (redoState != null)
        {
            ShowPdfPreview(redoState.PdfBytes);
            UpdateStatus($"已重做: {redoState.Description}");
        }
        
        LogHelper.Info($"[PdfOperationsPanel] 重做操作，剩余重做: {_historyManager.RedoCount}");
    }
    
    /// <summary>
    /// 更新历史按钮状态
    /// </summary>
    private void UpdateHistoryButtons()
    {
        _btnUndo.Enabled = _historyManager.CanUndo;
        _btnRedo.Enabled = _historyManager.CanRedo;
        
        // 更新按钮提示文本（可选）
        _btnUndo.Text = _historyManager.CanUndo 
            ? $"撤回 ({_historyManager.UndoCount})" 
            : "撤回";
        _btnRedo.Text = _historyManager.CanRedo 
            ? $"重做 ({_historyManager.RedoCount})" 
            : "重做";
    }
    
    /// <summary>
    /// 显示PDF预览
    /// </summary>
    private void ShowPdfPreview(byte[] pdfBytes)
    {
        string tempPath = Path.Combine(Path.GetTempPath(), 
            $"temp_preview_{Guid.NewGuid()}.pdf");
        File.WriteAllBytes(tempPath, pdfBytes);
        _ = _pdfPreview.LoadPdfAsync(tempPath);
    }
    
    /// <summary>
    /// 保存当前状态
    /// </summary>
    private void BtnSave_Click(object sender, EventArgs e)
    {
        if (!_historyManager.CanUndo)
        {
            ShowWarning("没有需要保存的更改");
            return;
        }
        
        // 获取当前状态（栈顶）
        var currentState = _historyManager.GetUndoHistory().FirstOrDefault();
        if (currentState != null)
        {
            var topItem = _historyManager.Undo();
            _historyManager.PushOperation(topItem); // 放回去
            
            File.WriteAllBytes(_originalFilePath, topItem.PdfBytes);
            
            // 保存后更新原始字节
            _originalPdfBytes = topItem.PdfBytes;
            
            // 清除历史
            _historyManager.Clear();
            UpdateHistoryButtons();
            
            ShowSuccess("文件已保存");
            UpdateStatus($"已保存: {Path.GetFileName(_originalFilePath)}");
        }
    }
}
```

### 方案二：简化历史栈系统（推荐当前实现）

考虑到当前主要是字体转曲操作，可以实现一个简化版本：

#### 核心改进点
1. **保存原始PDF字节** - 不依赖文件加载
2. **使用栈结构** - 简单的撤回/重做栈
3. **立即启用重做** - 撤回后重做按钮立即可用
4. **显示操作计数** - 按钮文本显示可撤回/重做的次数

#### 实现代码
```csharp
public partial class PdfOperationsPanel : BasePanelControl
{
    // 历史栈
    private Stack<byte[]> _undoStack = new Stack<byte[]>();
    private Stack<byte[]> _redoStack = new Stack<byte[]>();
    
    // 原始文件
    private string _originalFilePath;
    private byte[] _originalPdfBytes;
    
    // 当前显示的PDF
    private byte[] _currentPdfBytes;
    
    /// <summary>
    /// 加载PDF文件
    /// </summary>
    private async Task LoadPdfFileAsync(string filePath)
    {
        _originalFilePath = filePath;
        _originalPdfBytes = File.ReadAllBytes(filePath);
        _currentPdfBytes = _originalPdfBytes;
        
        // 清除历史
        _undoStack.Clear();
        _redoStack.Clear();
        UpdateHistoryButtons();
        
        await _pdfPreview.LoadPdfAsync(filePath);
    }
    
    /// <summary>
    /// 字体转曲完成
    /// </summary>
    private void Inspector_FontOutlineCompleted(object sender, FontOutlineCompletedEventArgs e)
    {
        // 将当前状态压入撤回栈
        _undoStack.Push(_currentPdfBytes);
        
        // 更新当前状态
        _currentPdfBytes = e.PdfBytes;
        
        // 清除重做栈（新操作清除重做历史）
        _redoStack.Clear();
        
        UpdateHistoryButtons();
        ShowPdfPreview(_currentPdfBytes);
        
        LogHelper.Info($"[PdfOperationsPanel] 转曲完成，历史栈深度: {_undoStack.Count}");
    }
    
    /// <summary>
    /// 撤回
    /// </summary>
    private void BtnUndo_Click(object sender, EventArgs e)
    {
        if (_undoStack.Count == 0)
        {
            ShowWarning("没有可以撤回的操作");
            return;
        }
        
        // 将当前状态压入重做栈
        _redoStack.Push(_currentPdfBytes);
        
        // 从撤回栈弹出上一个状态
        _currentPdfBytes = _undoStack.Pop();
        
        UpdateHistoryButtons();
        ShowPdfPreview(_currentPdfBytes);
        
        bool isOriginal = _undoStack.Count == 0;
        UpdateStatus(isOriginal 
            ? $"已加载: {Path.GetFileName(_originalFilePath)}"
            : $"已撤回 (未保存) - {Path.GetFileName(_originalFilePath)}");
        
        LogHelper.Info($"[PdfOperationsPanel] 撤回，剩余: {_undoStack.Count}, 可重做: {_redoStack.Count}");
    }
    
    /// <summary>
    /// 重做
    /// </summary>
    private void BtnRedo_Click(object sender, EventArgs e)
    {
        if (_redoStack.Count == 0)
        {
            ShowWarning("没有可以重做的操作");
            return;
        }
        
        // 将当前状态压入撤回栈
        _undoStack.Push(_currentPdfBytes);
        
        // 从重做栈弹出
        _currentPdfBytes = _redoStack.Pop();
        
        UpdateHistoryButtons();
        ShowPdfPreview(_currentPdfBytes);
        
        UpdateStatus($"已重做 (未保存) - {Path.GetFileName(_originalFilePath)}");
        
        LogHelper.Info($"[PdfOperationsPanel] 重做，历史: {_undoStack.Count}, 剩余重做: {_redoStack.Count}");
    }
    
    /// <summary>
    /// 更新历史按钮
    /// </summary>
    private void UpdateHistoryButtons()
    {
        _btnUndo.Enabled = _undoStack.Count > 0;
        _btnRedo.Enabled = _redoStack.Count > 0;
        
        // 显示计数（可选）
        _btnUndo.Text = _undoStack.Count > 0 ? $"撤回 ({_undoStack.Count})" : "撤回";
        _btnRedo.Text = _redoStack.Count > 0 ? $"重做 ({_redoStack.Count})" : "重做";
        
        // 保存按钮：有历史记录就可以保存
        _btnSave.Enabled = _undoStack.Count > 0 || _redoStack.Count > 0;
        _btnSaveAs.Enabled = true;
    }
    
    /// <summary>
    /// 显示PDF预览
    /// </summary>
    private void ShowPdfPreview(byte[] pdfBytes)
    {
        string tempPath = Path.Combine(Path.GetTempPath(), 
            $"temp_preview_{Guid.NewGuid()}.pdf");
        File.WriteAllBytes(tempPath, pdfBytes);
        _ = _pdfPreview.LoadPdfAsync(tempPath);
    }
    
    /// <summary>
    /// 保存
    /// </summary>
    private void BtnSave_Click(object sender, EventArgs e)
    {
        if (_currentPdfBytes == _originalPdfBytes)
        {
            ShowWarning("没有需要保存的更改");
            return;
        }
        
        try
        {
            File.WriteAllBytes(_originalFilePath, _currentPdfBytes);
            
            // 保存后更新原始字节并清除历史
            _originalPdfBytes = _currentPdfBytes;
            _undoStack.Clear();
            _redoStack.Clear();
            UpdateHistoryButtons();
            
            ShowSuccess("文件已保存");
            UpdateStatus($"已保存: {Path.GetFileName(_originalFilePath)}");
        }
        catch (Exception ex)
        {
            LogHelper.Error($"[PdfOperationsPanel] 保存失败: {ex.Message}", ex);
            ShowError($"保存失败: {ex.Message}");
        }
    }
}
```

## 内存管理建议

### PDF文件大小考虑
- 小文件（< 5MB）：可以保存 10-20 个历史状态
- 中等文件（5-20MB）：建议保存 5-10 个历史状态
- 大文件（> 20MB）：建议保存 3-5 个历史状态

### 优化策略
```csharp
private const int MAX_HISTORY_SIZE = 10;
private const long MAX_MEMORY_MB = 200; // 最大内存使用 200MB

private void LimitHistorySize()
{
    // 计算当前内存使用
    long totalMemory = _undoStack.Sum(x => x.Length) + 
                       _redoStack.Sum(x => x.Length);
    
    // 如果超过限制，移除最旧的历史
    while (_undoStack.Count > MAX_HISTORY_SIZE || 
           totalMemory > MAX_MEMORY_MB * 1024 * 1024)
    {
        if (_undoStack.Count > 1)
        {
            var items = _undoStack.ToList();
            items.RemoveAt(items.Count - 1);
            _undoStack = new Stack<byte[]>(items.Reverse<byte[]>());
            
            totalMemory = _undoStack.Sum(x => x.Length) + 
                         _redoStack.Sum(x => x.Length);
        }
        else
        {
            break;
        }
    }
}
```

## 总结

**推荐实现方案二（简化历史栈）**，因为：
1. 实现简单，易于维护
2. 满足当前需求（主要是字体转曲）
3. 性能良好，内存可控
4. 用户体验接近专业软件
5. 为未来扩展留有空间

这个方案可以让撤回/重做功能像 Adobe Acrobat DC 一样工作，同时保持代码简洁。
